async function fetchGlobalTraffic() { const debugNode = document.getElementById('debug-traffic'); // On construit l'URL avec uniquement les lignes qui nous intéressent const lineRefs = Object.keys(LINE_MAP).map(id => `LineRef=STIF:Line::${id}:`).join('&'); const url = `https://prim.iledefrance-mobilites.fr/marketplace/general-message?${lineRefs}`; try { // On supprime le proxy et on appelle l'API directement ! const response = await fetch(url, { headers: { 'apiKey': API_KEY, 'Accept': 'application/json' } }); if (!response.ok) { throw new Error(`Erreur HTTP ${response.status}`); } const data = await response.json(); const messages = data.Siri?.ServiceDelivery?.GeneralMessageDelivery?.[0]?.InfoMessage || []; let newQueue = [{text: DEFAULT_MSG, type: 'default'}]; messages.forEach(m => { const contentLineRefs = m.Content?.LineRef || []; const channel = m.InfoChannelRef?.value; contentLineRefs.forEach(lRef => { const idMatch = lRef.value.match(/C\d{5}/); if (!idMatch) return; const id = idMatch[0]; const lineName = LINE_MAP[id]; if (lineName) { const msgs = m.Content?.Message || []; const short = msgs.find(msg => msg.MessageType === "SHORT_MESSAGE"); let text = short ? short.MessageText.value : (msgs[0]?.MessageText?.value || ""); if (text) { const isRERA = (id === 'C01742'); const isIncident = (channel === "Perturbation"); const isWork = (channel === "PlannedWork"); if (isRERA || isIncident) { let type = isWork ? 'works' : 'incident'; const formattedText = `[${lineName}] ${text}`; newQueue.push({text: formattedText, type: type}); } } } }); }); displayQueue = Array.from(new Set(newQueue.map(a => JSON.stringify(a)))).map(a => JSON.parse(a)); debugNode.innerHTML = `API TRAFFIC OK (${displayQueue.length-1} alertes)`; } catch (e) { // On affiche l'erreur exacte pour faciliter le diagnostic debugNode.innerHTML = `API TRAFFIC : ERREUR (${e.message})`; } }
